var sprite_bbox_top = sprite_get_bbox_top(sprite_index)-sprite_get_yoffset(sprite_index);
var sprite_bbox_bottom = sprite_get_bbox_bottom(sprite_index)-sprite_get_yoffset(sprite_index);
var sprite_bbox_left = sprite_get_bbox_left(sprite_index)-sprite_get_xoffset(sprite_index);
var sprite_bbox_right = sprite_get_bbox_right(sprite_index)-sprite_get_xoffset(sprite_index);
	
//Horizontal
x += hspd;
	
var wall_x = instance_place(x+hspd,y,oWall);
if wall_x != noone {
	if hspd > 0 {//right
		x = wall_x.bbox_left-sprite_bbox_right-1;
	} else {//left
		x = wall_x.bbox_right-sprite_bbox_left;
	}
	hspd = 0;
}
	
//Vertical
y += vspd;
	
var wall_y = instance_place(x,y+vspd,oWall);
if wall_y != noone {
	if vspd > 0 {//down
		y = wall_y.bbox_top-sprite_bbox_bottom-1;
	} else {//up
		y = wall_y.bbox_bottom-sprite_bbox_top;
	}
	vspd = 0;
}
